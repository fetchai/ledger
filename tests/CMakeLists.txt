function(standard_test name file)
  list(REMOVE_AT ARGV 0) 
  
  add_executable(${name} "${name}.cpp")
  target_link_libraries(${name} ${PNG_LIBRARY})
  add_test(${name} ${name} ${ARGV})
endfunction()
